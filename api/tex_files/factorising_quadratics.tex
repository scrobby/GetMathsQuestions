  if data['x2coeffs_include3']:
    coeffs_list.append(3)

for ieqn in range(number_of_questions):

  x2coeff = random.choice(coeffs_list)
  arl = x2coeff*answers_range[0]
  arh = x2coeff*answers_range[1]
  sol1 = random.randint(arl, arh)
  sol2 = random.randint(answers_range[0], answers_range[1])
  x2coeff_list.append(x2coeff)
  sol2_list.append(sol2)
  sol1_list.append(sol1)

  b = (-1)*(x2coeff*sol2) - sol1
  equals = random.randint(equals_range[0], equals_range[1])
  c = sol2*sol1 + equals

  if b <0:
    bsign = " - "
  else:
    bsign = " + "
  if c < 0:
    csign = " - "
  else:
    csign = " + "

  qnum = ieqn+1

  print("\n" + str(qnum) + ") ")
  if x2coeff == 1:
    print(r'\textit{$x^2$}' + bsign + str(abs(b)) + r'\textit{$x$}' + csign + str(abs(c)) + " = " + str(equals))
  else:
    print(str(x2coeff) + r'\textit{$x^2$}' + bsign + str(abs(b)) + r'\textit{$x$}' + csign + str(abs(c)) + " = " + str(equals))

  for i in range(7):
    print(r'\textit{\newline}')
